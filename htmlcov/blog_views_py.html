10def blog_feed(request): 

11 """A view to render blog posts ordered by newest first with paginations""" 

12 

13 blog_post = BlogPost.objects.all().order_by('-created_on') 

14 blog_image = BlogImage.objects.all() 

15 total_post_number = blog_post.count() 

16 

17 category_dict = dict() 

18 for key in blog_post: 

19 if key.category in category_dict.keys(): 

20 category_dict[key.category] += 1 

21 else: 

22 category_dict[key.category] = 1 

23 

24 # Pagination 

25 paginator = Paginator(blog_post, 5) 

26 page = request.GET.get('page') 

27 

28 try: 

29 blog_post = paginator.page(page) 

30 except PageNotAnInteger: 

31 blog_post = paginator.page(1) 

32 except EmptyPage: 

33 blog_post = paginator.page(paginator.num_pages) 

34 

35 context = { 

36 'blog_post': blog_post, 

37 'blog_image': blog_image, 

38 'category_dict': category_dict, 

39 'total_post_number': total_post_number,
